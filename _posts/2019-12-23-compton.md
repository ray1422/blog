編輯`~/.config/compton.conf`加入以下行 ：
```bash
shadow-exclude = [
    "argb",
    "_NET_WM_OPAQUE_REGION@:c",
    "! name~=''",
    "g:e:Xfce4-notifyd"
]
wintypes:
{
    menu          = { shadow = false; };
    dropdown_menu = { shadow = false; };
    popup_menu    = { shadow = false; };
};
```
